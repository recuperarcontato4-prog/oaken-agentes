Codigo destaque

import pandas as pd
from openai import OpenAI

def auto_eda(csv_path: str) -> dict:
    """Gera EDA completa a partir de um CSV."""
    df = pd.read_csv(csv_path)
    stats = {
        "shape": df.shape,
        "dtypes": df.dtypes.to_dict(),
        "nulls": df.isnull().sum().to_dict(),
        "describe": df.describe().to_dict(),
        "correlations": df.select_dtypes("number").corr().to_dict(),
    }
    return stats

def generate_narrative(stats: dict) -> str:
    """Envia estatisticas ao LLM para narrativa em linguagem natural."""
    client = OpenAI()
    resp = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{
            "role": "system",
            "content": "Voce e um analista de dados senior. Gere um relatorio EDA."
        }, {
            "role": "user",
            "content": f"Estatisticas: {stats}"
        }]
    )
    return resp.choices[0].message.content

O que aprendi

  • EDA programatica — automatizar describe, corr e null-check garante consistencia em qualquer dataset.
  • Narrativa gerada por IA — LLM transforma numeros frios em insights acionaveis para stakeholders nao-tecnicos.
  • Graficos automaticos — Matplotlib com heuristicas de tipo (categorico vs numerico) escolhe o chart certo.
  • Pipeline reprodutivel — CSV in, HTML report out, sem intervencao manual.
