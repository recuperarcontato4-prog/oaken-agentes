Código destaque

from langfuse.decorators import observe, langfuse_context
from langfuse import Langfuse
from openai import OpenAI

langfuse = Langfuse()
client = OpenAI()

@observe()
def rag_pipeline(query: str) -> str:
    """Pipeline RAG com tracing automático."""
    docs = retrieve_docs(query)
    answer = generate_answer(query, docs)
    return answer

@observe(as_type="generation")
def generate_answer(query: str, docs: list) -> str:
    """Gera resposta — Langfuse captura tokens e latência."""
    response = client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role": "user", "content": prompt}],
    )
    # Score automático de fidelidade
    langfuse_context.score_current_trace(
        name="faithfulness",
        value=evaluate_faithfulness(response, docs),
    )
    return response.choices[0].message.content

# Avaliação batch offline
for trace in langfuse.get_traces(limit=100):
    score = auto_evaluate(trace.input, trace.output)
    langfuse.score(trace_id=trace.id,
                   name="quality", value=score)

O que aprendi

  • Observabilidade LLM — Tracing mostra exatamente onde o tempo e tokens são gastos em pipelines complexos.
  • Tracing hierárquico — Traces → Spans → Generations mapeiam chamadas aninhadas com zero boilerplate.
  • Avaliação automática — Scores por trace permitem monitorar qualidade em produção sem revisão manual.
  • Custo tracking — Dashboard de custo por modelo/feature evita surpresas na fatura da API.
